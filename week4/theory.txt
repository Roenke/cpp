static_cast<T2>(T1); // Приведение для примитивных типов, когда известно, что есть возможность перевести один тип в другой.
static_cast<void>(expr) // Валидный код. Означает отбросить результат

const_cast<T2>(T1); // Позволяет добавлять константность, volatile,
